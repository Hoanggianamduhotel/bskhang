import { Printer, Eye } from "lucide-react";
import { useToast } from "@/hooks/use-toast";

  ho_ten?: string;
  ngay_sinh?: string;
  thang_tuoi?: number;
  can_nang?: string;
  dia_chi?: string;
  so_dien_thoai?: string;
}

interface InToaThuocProps {
  khambenhID: string;
  disabled?: boolean;
  patientInfo?: PatientInfoType | null;
}

  tenBacSi: string;

  ngaySinh?: string;

  canNang?: string;
  diaChi?: string;
  soDienThoai?: string;

    don_vi: string;
    so_lan_dung: number;
    so_luong_moi_lan: number;
    tong_so_luong: number;
    ghi_chu?: string;
  }[];
}

export interface InToaThuocRef {
  handlePreview: () => void;
  handlePrint: () => void;
}
